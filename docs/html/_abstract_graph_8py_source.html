531  def getEvaluationOrder(self, node):
532 
533  order = {0: []}
534 
535  # include first node only if it is callable
536  if not node.bCallable:
537  order[0].append(node)
538 
539  def foo(n, process=True):
540  if not process:
541  return
542  next_layer_nodes = self.getNextLayerNodes(n, PinDirection.Input)
543 
544  layer_idx = max(order.iterkeys()) + 1
545  for n in next_layer_nodes:
546  if layer_idx not in order:
547  order[layer_idx] = []
548  order[layer_idx].append(n)
549  for i in next_layer_nodes:
550  foo(i)
551  foo(node)
552 
553  # make sure no copies of nodes in higher layers (non directional cycles)
554  for i in reversed(sorted([i for i in order.iterkeys()])):
555  for iD in range(i - 1, -1, -1):
556  for check_node in order[i]:
557  if check_node in order[iD]:
558  order[iD].remove(check_node)
559  return order
