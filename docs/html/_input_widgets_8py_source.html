530  return IntInputWidget(dataSetCallback=dataSetter, defaultValue=defaultValue)

532  return StringInputWidget(dataSetCallback=dataSetter, defaultValue=defaultValue)

538  return FloatVector4InputWidget(dataSetCallback=dataSetter, defaultValue=defaultValue)

542  return Matrix33InputWidget(dataSetCallback=dataSetter, defaultValue=defaultValue)
543  if dataType == DataTypes.Matrix44:
544  return Matrix44InputWidget(dataSetCallback=dataSetter, defaultValue=defaultValue)
545  if dataType == DataTypes.Exec:
