151  def match_multiline(self, text, delimiter, in_state, style):
152  # If inside triple-single quotes, start at 0
153  if self.previousBlockState() == in_state:
154  start = 0
155  add = 0
156  # Otherwise, look for the delimiter on this line
157  else:
158  start = delimiter.indexIn(text)
159  # Move past this match
160  add = delimiter.matchedLength()
161 
162  # As long as there's a delimiter match on this line...
163  while start >= 0:
164  # Look for the ending delimiter
165  end = delimiter.indexIn(text, start + add)
166  # Ending delimiter on this line?
167  if end >= add:
168  length = end - start + add + delimiter.matchedLength()
169  self.setCurrentBlockState(0)
170  # No; multi-line string
171  else:
172  self.setCurrentBlockState(in_state)
173  length = text.length() - start + add
174  # Apply formatting
175  self.setFormat(start, length, style)
176  # Look for the next match
177  start = delimiter.indexIn(text, start + length)
178 
179  # Return True if still inside a multi-line string, False otherwise
180  if self.currentBlockState() == in_state:
181  return True
182  else:
